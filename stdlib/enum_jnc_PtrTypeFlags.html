bitflag enum PtrTypeFlags
{
    Safe       = 0x0010000,
    Const      = 0x0020000,
    ReadOnly   = 0x0040000,
    CMut       = 0x0080000,
    Volatile   = 0x0100000,
    Event      = 0x0200000,
    DualEvent  = 0x0400000,
    Bindable   = 0x0800000,
    AutoGet    = 0x1000000,
    DualTarget = 0x2000000,
    _Dual      = ReadOnly | CMut | DualEvent | DualTarget,
};
