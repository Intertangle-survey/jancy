class sys.RegKey

class RegKey
{
    // construction

    construct();
    destruct();

    // methods

    void close();

    bool errorcode create(
        sys.RegStdKey parent,
        char const* path
        );

    bool errorcode create(
        sys.RegKey* parent,
        char const* path
        );

    bool errorcode open(
        sys.RegStdKey parent,
        char const* path,
        sys.RegKeyAccess access = RegKeyAccess.All
        );

    bool errorcode open(
        sys.RegKey* parent,
        char const* path,
        sys.RegKeyAccess access = RegKeyAccess.All
        );

    void const* errorcode read(
        char const* name,
        sys.RegValueType* type = null
        );

    uint32_t readDword(char const* name);
    char const* readString(char const* name);

    bool errorcode write(
        char const* name,
        sys.RegValueType type,
        void const* data,
        size_t size
        );

    bool errorcode writeDword(
        char const* name,
        uint32_t value
        );

    bool errorcode writeString(
        char const* name,
        char const* value
        );

    // aliases

    alias dispose = close;
};
