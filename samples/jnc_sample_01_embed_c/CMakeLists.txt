	)

	pch
	FILES
	${PCH_H}
	${PCH_C}
	)

#

include_directories (
	${JANCY_INC_DIR}
	)

link_directories (
	${AXL_LIB_DIR}
	${JANCY_LIB_DIR}
	${LLVM_LIB_DIR}
	)

add_executable (
	jnc_sample_01_embed_c
	${PCH_H}
	${PCH_C}
	${APP_H_LIST}
	${APP_C_LIST}
	${APP_JNC_LIST}
	)

axl_set_pch (
	jnc_sample_01_embed_c
	${PCH_H}
	${PCH_C}
	)

set_target_properties (
	jnc_sample_01_embed_c
	PROPERTIES
	FOLDER samples
	)

if (BUILD_JNC_DLL)
	add_dependencies (
		jnc_sample_01_embed_c
		jnc_dll
		)

	target_link_libraries (
		jnc_sample_01_embed_c
		${JANCY_DLL_NAME}
		)
else ()
	target_link_libraries (
		jnc_sample_01_embed_c
		jnc_api_static
		jnc_ct
		jnc_rt
		jnc_rtl
		jnc_std
		jnc_sys
		jnc_api_static
		)

	target_link_llvm_jit_libraries (jnc_sample_01_embed_c)
endif ()

target_link_libraries (
	jnc_sample_01_embed_c
	axl_zip
	axl_fsm
	axl_io
	axl_lex
	axl_core
	)

if (UNIX AND NOT APPLE)
	target_link_libraries (
		jnc_sample_01_embed_c
		rt
		)
endif ()

add_test (
	NAME jnc_sample_01_embed_c
	COMMAND
		$<TARGET_FILE:jnc_sample_01_embed_c>

	)
