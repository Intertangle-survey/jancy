	)

	jnc_Error.jnc.cpp

	)

	)

	jnc_String.jnc.cpp

	${JNC2CPP_PL}
	)

	jnc_HashTable.jnc.cpp
	jnc_HashTable.jnc

	)

	jnc_List.jnc.cpp
	jnc_List.jnc
	${JNC2CPP_PL}
	)
